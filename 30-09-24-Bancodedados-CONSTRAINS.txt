 CREATE DATABASE regra_constraint;

CREATE TABLE usuario(
id_cliente INT AUTO_INCREMENT,
nome VARCHAR (50) NOT NULL,
cpf BIGINT UNIQUE,
pais VARCHAR (50) DEFAULT 'Brasil',
graduado ENUM('sim','nao'),
num_filhos INT,
CHECK (num_filhos >=0),
PRIMARY KEY(id_cliente)
);
